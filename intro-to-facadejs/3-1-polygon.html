Canvas Examples

3.1 Polygon

var canvas = document.querySelector('canvas'),
    context = canvas.getContext('2d');

context.beginPath();

context.moveTo(150, 250);
context.lineTo(250, 150);
context.lineTo(350, 250);
context.lineTo(250, 350);

context.closePath();

context.fillStyle = '#333';

context.fill();
