Canvas Examples

2.1 Circle With Stroke

var canvas = document.querySelector('canvas'),
    context = canvas.getContext('2d');

context.beginPath();

context.arc(250, 250, 50, 0, Math.PI * 180);

context.closePath();

context.fillStyle = '#333';

context.fill();

context.lineWidth = 10;
context.strokeStyle = '#33CC77';

context.stroke();
