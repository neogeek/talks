7.2 Sample Using Facade.js

var stage = new Facade(document.querySelector('canvas')),
    rect = new Facade.Rect({
        x: stage.width() / 2,
        y: stage.height() / 2,
        width: 100,
        height: 100,
        fillStyle: '#333',
        strokeStyle: '#33CC77',
        lineWidth: 10,
        scale: 2,
        rotate: 45,
        anchor: 'center'
    });

stage.addToStage(rect);
